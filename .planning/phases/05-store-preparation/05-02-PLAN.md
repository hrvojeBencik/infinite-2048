Create `lib/features/subscription/data/services/subscription_service.dart`:
```dart
import 'dart:io';
import 'package:flutter/foundation.dart';
import 'package:purchases_flutter/purchases_flutter.dart';
import '../../../../core/constants/app_constants.dart';

class SubscriptionService {
  bool _initialized = false;

  Future<void> initialize() async {
    if (_initialized) return;
    try {
      final apiKey = Platform.isIOS
          ? AppConstants.revenueCatAppleApiKey
          : AppConstants.revenueCatGoogleApiKey;
      final config = PurchasesConfiguration(apiKey);
      await Purchases.configure(config);
      _initialized = true;
    } catch (e) {
      debugPrint('RevenueCat init failed: $e');
    }
  }

  Future<Offerings?> getOfferings() async {
    if (!_initialized) return null;
    try {
      return await Purchases.getOfferings();
    } catch (e) {
      debugPrint('Failed to get offerings: $e');
      return null;
    }
  }

  Future<CustomerInfo?> purchasePackage(Package package) async {
    try {
      return await Purchases.purchasePackage(package);
    } catch (e) {
      debugPrint('Purchase failed: $e');
      rethrow;
    }
  }

  Future<CustomerInfo?> restorePurchases() async {
    try {
      return await Purchases.restorePurchases();
    } catch (e) {
      debugPrint('Restore failed: $e');
      rethrow;
    }
  }

  Future<bool> isPremium() async {
    if (!_initialized) return false;
    try {
      final info = await Purchases.getCustomerInfo();
      return info.entitlements.active.containsKey(AppConstants.premiumEntitlementId);
    } catch (e) {
      return false;
    }
  }
}
```

**Step 5: Create SubscriptionState**
Create `lib/features/subscription/presentation/bloc/subscription_state.dart`:
```dart
import 'package:equatable/equatable.dart';
import 'package:purchases_flutter/purchases_flutter.dart';

enum SubscriptionStatus { initial, loading, loaded, purchasing, error }

class SubscriptionState extends Equatable {
  final SubscriptionStatus status;
  final bool isPremium;
  final Offerings? offerings;
  final String? errorMessage;

  const SubscriptionState({
    this.status = SubscriptionStatus.initial,
    this.isPremium = false,
    this.offerings,
    this.errorMessage,
  });

  SubscriptionState copyWith({
    SubscriptionStatus? status,
    bool? isPremium,
    Offerings? offerings,
    String? errorMessage,
  }) {
    return SubscriptionState(
      status: status ?? this.status,
      isPremium: isPremium ?? this.isPremium,
      offerings: offerings ?? this.offerings,
      errorMessage: errorMessage,
    );
  }

  @override
  List<Object?> get props => [status, isPremium, offerings, errorMessage];
}
```

**Step 6: Create SubscriptionBloc**
Create `lib/features/subscription/presentation/bloc/subscription_bloc.dart`:
```dart
import 'package:flutter_bloc/flutter_bloc.dart';
import '../../data/services/subscription_service.dart';
import 'subscription_event.dart';
import 'subscription_state.dart';

class SubscriptionBloc extends Bloc<SubscriptionEvent, SubscriptionState> {
  final SubscriptionService _service;

  SubscriptionBloc({required SubscriptionService service})
      : _service = service,
        super(const SubscriptionState()) {
    on<SubscriptionCheckRequested>(_onCheckRequested);
    on<SubscriptionLoadOfferings>(_onLoadOfferings);
    on<SubscriptionPurchaseRequested>(_onPurchaseRequested);
    on<SubscriptionRestoreRequested>(_onRestoreRequested);
  }

  Future<void> _onCheckRequested(
    SubscriptionCheckRequested event,
    Emitter<SubscriptionState> emit,
  ) async {
    final isPremium = await _service.isPremium();
    emit(state.copyWith(isPremium: isPremium));
  }

  Future<void> _onLoadOfferings(
    SubscriptionLoadOfferings event,
    Emitter<SubscriptionState> emit,
  ) async {
    emit(state.copyWith(status: SubscriptionStatus.loading));
    final offerings = await _service.getOfferings();
    emit(state.copyWith(
      status: SubscriptionStatus.loaded,
      offerings: offerings,
    ));
  }

  Future<void> _onPurchaseRequested(
    SubscriptionPurchaseRequested event,
    Emitter<SubscriptionState> emit,
  ) async {
    emit(state.copyWith(status: SubscriptionStatus.purchasing));
    try {
      await _service.purchasePackage(event.package);
      final isPremium = await _service.isPremium();
      emit(state.copyWith(
        status: SubscriptionStatus.loaded,
        isPremium: isPremium,
      ));
    } catch (e) {
      emit(state.copyWith(
        status: SubscriptionStatus.error,
        errorMessage: 'Purchase could not be completed. Check your payment method and try again.',
      ));
    }
  }

  Future<void> _onRestoreRequested(
    SubscriptionRestoreRequested event,
    Emitter<SubscriptionState> emit,
  ) async {
    emit(state.copyWith(status: SubscriptionStatus.loading));
    try {
      await _service.restorePurchases();
      final isPremium = await _service.isPremium();
      emit(state.copyWith(
        status: SubscriptionStatus.loaded,
        isPremium: isPremium,
        errorMessage: isPremium ? null : 'No active subscription found for this Apple ID.',
      ));
    } catch (e) {
      emit(state.copyWith(
        status: SubscriptionStatus.error,
        errorMessage: 'No active subscription found for this Apple ID.',
      ));
    }
  }
}
```
